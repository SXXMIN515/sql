4.
bbb 테이블에 1부터 10까지 짝수 최종 합계 값을 PL/SQL 블록으로 작성하여 입력하시오.
단, if문 사용
힌트)
IF MOD(i,2) = 0 THEN
~

DECLARE
  v_sum  NUMBER :=0;
BEGIN
  FOR i in 1..10 LOOP
    IF MOD(i,2) = 0 THEN
      v_sum := v_sum+i;
    END IF;
  END LOOP;
  INSERT INTO bbb(b)

5.
1부터 10까지에서 짝수의 합계는 aaa 테이블에, 홀수의 합계는 bbb 테이블에 
입력되도록 PL/SQL 블록을 작성하시오. (단, if 문 사용)
DECLARE
  e_sum  NUMBER :=0;
  o_sum  NUMBER :=0;
BEGIN
  FOR i in 1..10 LOOP
    IF MOD(i,2) = 0 THEN
      e_sum := e_sum+i;
    ELSE
      o_sum := o_sum+i;
    END IF;
  END LOOP;
  INSERT INTO aaa(a)
  VALUES (e_sum);
  INSERT INTO bbb(b)
  VALUES (o_sum);
END;
